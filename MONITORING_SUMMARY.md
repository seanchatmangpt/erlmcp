### Add Custom Alert Rule
```erlang
CustomRule = #{
    id => <<"high_cpu">>,
    name => <<"High CPU Usage">>,
    condition => fun(Metrics) ->
        CpuUsage = maps:get(cpu_usage_percent, Metrics, 0),
        CpuUsage > 80.0
    end,
    threshold => 80.0,
    severity => warning,
    cooldown => 300000,
    enabled => true
},
erlmcp_monitor:add_alert_rule(<<"high_cpu">>, CustomRule).
```
