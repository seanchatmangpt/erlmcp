## Quick Start

```erlang
%% Start MCP server
{ok, Server} = erlmcp_server:start_link({stdio, []}, #{}),

%% Add a tool
erlmcp_server:add_tool(Server, <<"greet">>,
    fun(#{<<"name">> := Name}) ->
        <<"Hello, ", Name/binary, "!">>
    end).

%% Initialize
erlmcp_server:initialize(Server, Capabilities).
```
